CREATE TABLE `tf_article` (
	`id` INT(11) UNSIGNED NOT NULL AUTO_INCREMENT,
	`catid` INT(10) UNSIGNED NOT NULL DEFAULT '0',
	`title` VARCHAR(200) NOT NULL COMMENT '标题',
	`subtitle` VARCHAR(200) NOT NULL COMMENT '副标题',
	`content` TEXT NOT NULL COMMENT '内容',
	`thumb` VARCHAR(250) NOT NULL DEFAULT '' COMMENT '缩略图',
	`mainPic` VARCHAR(250) NOT NULL DEFAULT '' COMMENT '主图',
	`teacher` VARCHAR(50) NOT NULL DEFAULT '' COMMENT '讲师',
	`address` VARCHAR(200) NOT NULL DEFAULT '' COMMENT '地址',
	`number` INT(6) UNSIGNED NOT NULL DEFAULT '0' COMMENT '参与人数',
	`total` INT(6) UNSIGNED NOT NULL DEFAULT '0' COMMENT '总人数',
	`stime` INT(11) UNSIGNED NOT NULL DEFAULT '0' COMMENT '开始时间',
	`etime` INT(11) UNSIGNED NOT NULL DEFAULT '0' COMMENT '结束时间',
	`sort` INT(10) NOT NULL DEFAULT '0' COMMENT '排序',
	`status` TINYINT(2) NOT NULL DEFAULT '0' COMMENT '0=屏蔽 1=正常 -1=删除',
	`tel` CHAR(50) NOT NULL,
	`addtime` INT(11) UNSIGNED NOT NULL DEFAULT '0' COMMENT '发布时间',
	PRIMARY KEY (`id`),
	INDEX `catid` (`catid`),
	INDEX `status` (`status`)
)
COMMENT='文章表'
COLLATE='utf8_general_ci'
ENGINE=MyISAM
AUTO_INCREMENT=33
;

CREATE TABLE `tf_salon_enroll` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`member_id` INT(11) NOT NULL DEFAULT '0' COMMENT '会员id',
	`name` VARCHAR(50) NOT NULL COMMENT '姓名',
	`mobile` INT(11) NOT NULL COMMENT '联系电话',
	`baby` TINYINT(1) NOT NULL DEFAULT '0' COMMENT '是否有宝宝 1=有 0=没有',
	`age` SMALLINT(4) NOT NULL DEFAULT '0' COMMENT '多少周岁',
	`addtime` INT(11) NOT NULL DEFAULT '0' COMMENT '添加时间',
	PRIMARY KEY (`id`)
)
COMMENT='沙龙报名表'
COLLATE='utf8_general_ci'
ENGINE=MyISAM
AUTO_INCREMENT=3
;
